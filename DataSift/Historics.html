def prepare(hash, start, end_time, name, sources = 'twitter', sample = 100)
  params = {
    :hash => hash,
    :start => start,
    :end => end_time,
    :name => name,
    :sources => sources,
    :sample => sample
  }
  requires params
  DataSift.request(:POST, 'historics/prepare', @config, params)
end
