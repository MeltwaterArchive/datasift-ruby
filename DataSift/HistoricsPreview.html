def create(hash, sources, parameters, start, end_time = nil)
  params = {
    :hash       => hash,
    :sources    => sources,
    :parameters => parameters,
    :start      => start
  }
  requires params
  params.merge!(:end => end_time) unless end_time.nil?

  DataSift.request(:POST, 'preview/create', @config, params)
end
